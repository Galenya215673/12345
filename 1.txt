import sympy as sp

# Определяем переменные
x = sp.symbols('x')

# Задаем функцию, для которой будем находить производную
f = x**3 + 2*x**2 + 3*x + 4

# Вычисляем первую производную функции f(x)
df_dx = sp.diff(f, x)

# Выводим результат
print("Исходная функция:", f)
print("Дифференциал функции:", df_dx)
